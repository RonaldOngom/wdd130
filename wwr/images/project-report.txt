def compute_molar_mass(symbol_quantity_list, periodic_table_dict):
    total_mass = 0.0
    for symbol, quantity in symbol_quantity_list:
        if symbol not in periodic_table_dict:
            raise ValueError(f"Element '{symbol}' not in periodic table.")
        total_mass += periodic_table_dict[symbol][1] * quantity
    return total_mass

def compute_total_protons(symbol_quantity_list, periodic_table_dict):
    total_protons = 0
    for symbol, quantity in symbol_quantity_list:
        total_protons += periodic_table_dict[symbol][2] * quantity
    return total_protons

def compute_moles(sample_mass, molar_mass):
    if molar_mass <= 0:
        raise ValueError("Molar mass must be greater than zero.")
    return sample_mass / molar_mass

def main():
    try:
        formula = input("Enter molecular formula (e.g., H2O): ").strip()
        sample_mass = float(input("Enter mass in grams: "))
        if sample_mass <= 0:
            print("Sample mass must be positive.")
            return

        periodic_table = make_periodic_table()
        symbol_quantity_list = [(s, q) for s, q in parse_formula(formula)]

        molar_mass = compute_molar_mass(symbol_quantity_list, periodic_table)
        moles = compute_moles(sample_mass, molar_mass)

        print(f"Total number of protons in molecule: {total_protons}")

    except Exception as e:
        print(f"Error: {e}")

if __name__ == "__main__":
    main()

# formula.py
# Parses chemical formulas into (symbol, quantity) tuples

import re

def parse_formula(formula):
    pattern = r'([A-Z][a-z]?)(\d*)'
    matches = re.findall(pattern, formula)
    if not matches:
        raise ValueError("Invalid formula.")
    result = []
    for symbol, qty in matches:
        quantity = int(qty) if qty else 1
        result.append((symbol, quantity))
    return result

5. `compute_moles(sample_mass, molar_mass)` - Calculates the number of moles from the sample mass and molar mass.
